# Host is dead (before the hunt, or from disease)
unop.0080 = {
	type = character_event
	title = hunt.0080.t
	desc = hunt.0080.death
	left_portrait = {
		character = root
		animation = worry
	}
	right_portrait = {
		character = scope:host
		triggered_animation = {
			trigger = { is_alive = no }
			animation = loss_1
		}
		animation = stress
	}
	theme = hunt_activity
	cooldown = { weeks = 1 }

	immediate = {
		if = {
			limit = { NOT = { this = scope:host } }
			hunt_invalidation_event_effect = yes
		}
	}

	option = {
		name = hunt.0080.a
		custom_tooltip = hunt_ends_tt
	}
}
