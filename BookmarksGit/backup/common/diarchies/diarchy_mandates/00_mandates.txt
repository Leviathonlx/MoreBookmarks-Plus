fill_coffers = {
	qualification_score = {
		add = {
			desc = mandate_aptitude.fill_coffers.category
			# Primary skill.
			add = {
				value = stewardship
				desc = mandate_aptitude.fill_coffers.primary_skill.stewardship
			}
			# Secondary skills.
			add = {
				value = learning
				multiply = 0.5
				desc = mandate_aptitude.fill_coffers.secondary_skill.learning
			}
			add = {
				value = intrigue
				multiply = 0.5
				desc = mandate_aptitude.fill_coffers.secondary_skill.intrigue
			}
			# Bonus traits.
			if = {
				limit = { has_trait = arbitrary }
				add = {
					value = 20
					desc = mandate_aptitude.fill_coffers.bonus_trait.arbitrary
				}
			}
			if = {
				limit = { has_trait = callous }
				add = {
					value = 20
					desc = mandate_aptitude.fill_coffers.bonus_trait.callous
				}
			}
			# Malus traits.
			if = {
				limit = { has_trait = generous }
				add = {
					value = -20
					desc = mandate_aptitude.fill_coffers.malus_trait.generous
				}
			}
			if = {
				limit = { has_trait = greedy }
				add = {
					value = -20
					desc = mandate_aptitude.fill_coffers.malus_trait.greedy
				}
			}
		}
	}

	ai_score = {
		value = 1
	}

promote_merit = {
	qualification_score = {
		add = {
			desc = mandate_aptitude.promote_merit.category
			# Primary skill.
			add = {
				value = merit_level
				multiply = 2
				desc = mandate_aptitude.promote_merit.merit_level
			}
			# Secondary skills.
			add = {
				value = 0
				every_relation = {
					type = disciple
					add = 1
				}
				desc = mandate_aptitude.promote_merit.disciples
			}
			add = {
				value = "has_trait_xp(confucian_education)"
				multiply = 0.25
				desc = mandate_aptitude.promote_merit.confucian_education
			}
		}
		if = {
			limit = {
				this = title:e_minister_grand_marshal.holder
			}
			multiply = 0
		}
	}


	ai_score = {
		value = 0
		if = {
			limit = {
				hegemon_favored_stability_phase_value >= 0
			}
			add = 10
		}
		if = {
			limit = {
				has_realm_law = grand_chancellor_law
			}
			add = 10
		}
		if = {
			limit = {
				diarch ?= {
					OR = {
						var:diarch_forced_mandate ?= flag:placate_movements
						var:diarch_forced_mandate ?= flag:promote_might
					}
				}
			}
			multiply = 0
		}
		if = {
			limit = {
				diarch ?= {
					var:diarch_forced_mandate ?= flag:promote_merit
				}
			}
			add = 100
		}
	}
}
