no_aspect = {
	is_default = yes
	show_in_main_hud = yes

	is_shown = {
		house_head ?= { 
			government_has_flag = government_is_mandala
			has_mandala_aspect_trigger = no 
		}
	}
	cooldown = { years = 0 }


	level = {
		parameters = {
			no_aspect = yes
		}
		ai_score = {
			value = 0
			if = {
				limit = {
					months_from_game_start < 1
					highest_held_title_tier < tier_kingdom
				}
				add = 1000
			}
		}
	}

	illustration = "gfx/interface/icons/culture_pillars/ethos_communal.dds"
}
