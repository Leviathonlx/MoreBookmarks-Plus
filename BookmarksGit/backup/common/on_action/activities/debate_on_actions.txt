debate_prep_opening_on_action = {
	effect = {
		involved_activity = {
			every_attending_character = {
				limit = { location = involved_activity.activity_location }
				trigger_event = debate_event.1000
			}
		}
	}
}

debate_opening_on_action = {
	effect = {
		involved_activity = {
			every_attending_character = {
				limit = { location = involved_activity.activity_location }
				if = {
					limit = {
						OR = {
							this = scope:host
							this = scope:activity.var:challenged_movement_leader
						}
					}
					trigger_event = debate_event.1010
				}
				else = {
					trigger_event = debate_event.1015
				}
			}
		}
	}
}
