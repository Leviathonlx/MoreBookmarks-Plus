# Petition Liege On Action from Activity Rework
tribute_mission_start = {
	effect = {
		add_character_flag = tribute_mission_character_flag
	}
}

tribute_mission_travel_planner_exit = {
	effect = {
		#We've actually moved somewhere
		if = {
			limit = {
				OR = {
					has_variable = offered_gold_value
					has_variable = offered_herd_value
				}
			}
			send_interface_toast = {
				type = event_toast_effect_neutral
				title = tribute_mission_costs_refunded
				desc = tribute_mission_costs_refunded
				left_icon = root
				#Gold: refunded
				if = {
					limit = { has_variable = offered_gold_value }
					add_gold = var:offered_gold_value
				}
				#Herd: refunded
				if = {
					limit = { has_variable = offered_herd_value }
					domicile = { change_herd = var:offered_herd_value }
				}
			}
		}

		#And clean up the variables
		tribute_mission_clean_up_variables_effect = yes

	}
}
