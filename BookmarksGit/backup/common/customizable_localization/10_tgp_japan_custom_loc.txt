GetImperialFamilyType = {
	type = character

	text = {
		trigger = { tgp_is_ceremonial_regent_trigger = yes } # Not 'actual' ruler
		localization_key = game_concept_regent_family
	}

	text = {
		trigger = { tgp_is_ceremonial_regent_trigger = no }
		localization_key = game_concept_imperial_family
	}
}

TopLiegeIsRegentOrImperialFamilyType = { 
	type = character
	text = {
		trigger = { 
			OR = {
				AND = {
					is_independent_ruler = yes 
					tgp_is_ceremonial_regent_trigger = yes
				}
				AND = {
					is_independent_ruler = no 
					tgp_realm_has_ceremonial_liege_trigger = yes
				}
			}
		}
		localization_key = FAMILY_STATUS_REGENT
	}

	text = {
		trigger = { 
			OR = {
				AND =  {
					is_independent_ruler = yes 
					tgp_is_ceremonial_regent_trigger = no
				}
				AND = {
					is_independent_ruler = no 
					tgp_realm_has_ceremonial_liege_trigger = no
				}
			}
		}
		localization_key = FAMILY_STATUS_IMPERIAL
	}
}
