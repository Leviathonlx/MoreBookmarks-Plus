# Nephew - siostrzeniec/bratanek
# Niece - siostrzenica/bratanica
PL_nephew = {
	log_loc_errors = no
	type = character

	text = {
		localization_key = nephew_from_brother
		trigger = {
				root.father = {
					is_sibling_of = scope:second
			}

		}
	}

	text = {
		fallback = yes
		localization_key = nephew_from_sister
		trigger = {
				root.mother = {
					is_sibling_of = scope:second
				}
		}
	}
}
