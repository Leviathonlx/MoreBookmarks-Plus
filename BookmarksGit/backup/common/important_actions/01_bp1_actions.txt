action_ongoing_house_feud = {
	priority = 300
	combine_into_one = yes
	is_dangerous = yes
	
	check_create_action = {
		if = {
			limit = {
				house = {
					save_temporary_scope_as = my_house_temp
					any_house_relation = {
						has_house_relation_level = feud
						any_relation_house = {
							NOT = { this = scope:my_house_temp }
							house_head ?= { save_temporary_scope_as = house_feud_rival_temp }
						}
					}
				}
			}
			try_create_important_action = {
				important_action_type = action_ongoing_house_feud
				actor = root
				recipient = scope:house_feud_rival_temp
			}
		}
	}

	effect = {
		scope:recipient.house = {
			open_view_data = {
				view = dynasty_house_view
			}
		}
	}
}
