action_reactive_advice_welcome_to_japan = {
	type = tutorial
	icon = generic_reactive_advice
	soundeffect = "event:/SFX/UI/Notifications/TopBar/sfx_ui_notifications_topbar_reactive_advice"
	check_create_action = {
		if = {
			limit = {
				should_show_tutorial_alert = { LESSON = reactive_advice_welcome_to_japan }
				OR = {
					has_government = japan_administrative_government
					has_government = japan_feudal_government
				}
				
			}
			try_create_important_action = {
				important_action_type = action_reactive_advice_welcome_to_japan
				actor = root
			}
		}
	}
	effect = {
		start_tutorial_lesson = reactive_advice_welcome_to_japan
	}
}

# grow your bloc

action_reactive_advice_lead_your_soryo_bloc = {
	type = tutorial
	icon = generic_reactive_advice
	soundeffect = "event:/SFX/UI/Notifications/TopBar/sfx_ui_notifications_topbar_reactive_advice"
	check_create_action = {
		if = {
			limit = {
				should_show_tutorial_alert = { LESSON = reactive_advice_lead_your_soryo_bloc }
				has_government = japan_feudal_government
				is_tutorial_lesson_completed = reactive_advice_soryo_intro

				# is bloc leader, but not ruling japan
				house.house_confederation.leading_house.house_head ?= this
				is_independent_ruler = no
			}
			try_create_important_action = {
				important_action_type = action_reactive_advice_lead_your_soryo_bloc
				actor = root
			}
		}
	}
	effect = {
		start_tutorial_lesson = reactive_advice_lead_your_soryo_bloc
	}
}
