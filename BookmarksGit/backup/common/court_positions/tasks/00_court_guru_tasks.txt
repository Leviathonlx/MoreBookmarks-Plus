guru_teach_learning = {
	court_position_types = { court_guru_court_position }
	
	is_valid_showing_failures_only = {
		learning >= 12
	}

	cost = {
		round = no
		prestige = {
			add = {
				value = monthly_court_position_task_cost_triple
				desc = COURT_POSITION_TASK_COST_BREAKDOWN_BASE
				format = "BASE_VALUE_FORMAT_DECIMALS_PLUS_NEGATIVE"
			}
		}
	}
	
	base_employer_modifier = {
		learning = 1
		stress_gain_mult = 0.2
	}
	
	scaling_employer_modifiers = {
		terrible = {
		 	monthly_learning_lifestyle_xp_gain_add = 2
		}
		poor = {
		 	monthly_learning_lifestyle_xp_gain_add = 4
		}
		average = {
		 	monthly_learning_lifestyle_xp_gain_add = 6
		}
		good = {
		 	monthly_learning_lifestyle_xp_gain_add = 10
		}
		excellent = {
		 	monthly_learning_lifestyle_xp_gain_add = 15
		}
	}
	
	on_start = { }
	
	on_end = { }

	ai_will_do = {
		value = 100
		if = {
			limit = {
				has_trait = theologian
				has_trait = scholar
				has_trait = whole_of_body
			}
			add = -500
		}
	}
}
