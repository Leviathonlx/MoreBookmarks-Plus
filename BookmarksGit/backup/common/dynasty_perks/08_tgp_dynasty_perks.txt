#Southeast Asia
tgp_sea_legacy_1 = {
	legacy = tgp_southeast_asia_legacy_track

	#Travel Safety
	#Legitimacy from Pilgrimages and Visiting Capital Temples

	can_be_picked = {
		has_tgp_dlc_trigger = yes
		OR = { #Can be accessed by both Mandala and Wanua, useful mostly for those Wanuas that want to become Mandalas
			government_has_flag = government_is_mandala
			government_has_flag = government_is_wanua
		}
	}

	character_modifier = {
		character_travel_safety = 10
	}

	effect = {
		custom_description_no_bullet = { text = tgp_legitimacy_visit_capital_temples }
		custom_description_no_bullet = { text = tgp_legitimacy_pilgrimages }
	}

	ai_chance = {
		value = 12
		if = {
			limit = {
				OR = {
					government_has_flag = government_is_mandala
					government_has_flag = government_is_wanua
				}
			}
			multiply = 10
		}
		if = {
			limit = {
				can_start_new_legacy_track_trigger = no
			}
			multiply = 0
		}
	}
}
