
tgp_uses_house_blocs_trigger = {
	has_tgp_dlc_trigger = yes
	government_has_flag = government_has_house_blocs # DO NOT CHANGE THIS
	top_liege = { government_has_flag = government_has_house_blocs } # DO NOT CHANGE THIS
}

tgp_is_valid_member_character_shared_trigger = {
	leading_house.house_head.top_liege ?= scope:character.top_liege # Same realm as Leading House head
	any_confederation_member_house = { this = scope:character.house }
	scope:character = { tgp_uses_house_blocs_trigger = yes }
}

tgp_is_valid_member_house_shared_trigger = {
	scope:house.house_head ?= {
		is_alive = yes
		any_held_title = { is_noble_family_title = yes } # Members must be Noble Families
		tgp_uses_house_blocs_trigger = yes
	}
}
