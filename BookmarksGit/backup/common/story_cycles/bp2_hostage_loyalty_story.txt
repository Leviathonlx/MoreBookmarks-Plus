bp2_hostage_loyalty = {
	# by James Beaumont
	
	on_setup = {
		
	}
#
	on_end = {
		debug_log = "Hostage loyalty story ended on:"
		debug_log_date = yes
	}
#
	on_owner_death = {
		end_story = yes
	}

	# Hostage dies or ceases to be a hostage, end the story
	effect_group = {
		days = { 40 60 }

		trigger = {
			exists = var:hostage_target
		}

		first_valid = {
			triggered_effect = {
				trigger = {
					exists = var:hostage_target
					var:hostage_target = {
						OR = {
							is_alive = no
							AND = {
								is_adult = no
								NOT = { is_hostage_of = root.story_owner }
							}
						}
					}
				}
				effect = {
					end_story = yes
				}
			}
		}
	}

	# Hostage comes of age, calculate final outcome
	effect_group = {
		days = { 1 3 }

		trigger = {
			exists = var:hostage_target
			var:hostage_target = { is_adult = yes }
		}

		first_valid = {
			triggered_effect = {
				trigger = { always = yes }
				effect = {
					save_scope_as = hostage_story
					var:hostage_target = {
						save_scope_as = hostage
					}
					random_list = {
						10 = { # Positive outcome, adopt the hostage
							modifier = {
								exists = var:theon_points
								add = {
									value = var:theon_points
									multiply = 2
								}
							}
							modifier = {
								story_owner = { has_relation_friend = scope:hostage }
								add = 5
							}
							modifier = {
								story_owner = {
									any_child = { has_relation_friend = scope:hostage }
								}
								add = 5
							}
							modifier = {
								story_owner = {
									opinion = {
										target = scope:hostage
										value >= high_positive_opinion
									}
								}
								add = 5
							}
							modifier = {
								story_owner = {
									opinion = {
										target = scope:hostage
										value >= low_positive_opinion
									}
								}
								add = 5
							}
							modifier = {
								story_owner = { has_relation_rival = scope:hostage }
								add = -10
							}
							modifier = {
								story_owner = {
									any_child = { has_relation_rival = scope:hostage }
								}
								add = -5
							}
							modifier = {
								story_owner = {
									opinion = {
										target = scope:hostage
										value >= high_negative_opinion
									}
								}
								add = -5
							}
							modifier = {
								story_owner = {
									opinion = {
										target = scope:hostage
										value >= low_negative_opinion
									}
								}
								add = -5
							}
							story_owner = { trigger_event = bp2_yearly.8060 }
						}
						20 = { # Positive outcome, make the hostage loyal
							modifier = {
								exists = var:theon_points
								add = var:theon_points
							}
							modifier = {
								story_owner = { has_relation_friend = scope:hostage }
								add = 2
							}
							modifier = {
								story_owner = {
									any_child = { has_relation_friend = scope:hostage }
								}
								add = 2
							}
							modifier = {
								story_owner = {
									opinion = {
										target = scope:hostage
										value >= high_positive_opinion
									}
								}
								add = 5
							}
							modifier = {
								story_owner = {
									opinion = {
										target = scope:hostage
										value >= low_positive_opinion
									}
								}
								add = 5
							}
							modifier = {
								story_owner = { has_relation_rival = scope:hostage }
								add = -5
							}
							modifier = {
								story_owner = {
									any_child = { has_relation_rival = scope:hostage }
								}
								add = -5
							}
							modifier = {
								story_owner = {
									opinion = {
										target = scope:hostage
										value >= high_negative_opinion
									}
								}
								add = -5
							}
							modifier = {
								story_owner = {
									opinion = {
										target = scope:hostage
										value >= low_negative_opinion
									}
								}
								add = -5
							}
							story_owner = { trigger_event = bp2_yearly.8061 }
						}
						20 = { # Neutral outcome, nothing happens
							end_story = yes
						}
						10 = { # Negative outcome, hostage now has rivalry with you
							modifier = {
								exists = var:theon_points
								add = {
									value = var:theon_points
									multiply = -1
								}
							}
							modifier = {
								story_owner = { has_relation_friend = scope:hostage }
								add = -20
							}
							modifier = {
								story_owner = {
									any_child = { has_relation_friend = scope:hostage }
								}
								add = -5
							}
							modifier = {
								story_owner = {
									opinion = {
										target = scope:hostage
										value >= high_positive_opinion
									}
								}
								add = -5
							}
							modifier = {
								story_owner = {
									opinion = {
										target = scope:hostage
										value >= low_positive_opinion
									}
								}
								add = -5
							}
							modifier = {
								story_owner = { has_relation_rival = scope:hostage }
								add = 20
							}
							modifier = {
								story_owner = {
									any_child = { has_relation_rival = scope:hostage }
								}
								add = 5
							}
							modifier = {
								story_owner = {
									opinion = {
										target = scope:hostage
										value >= high_negative_opinion
									}
								}
								add = 5
							}
							modifier = {
								story_owner = {
									opinion = {
										target = scope:hostage
										value >= low_negative_opinion
									}
								}
								add = 5
							}
							story_owner = { trigger_event = bp2_yearly.8063 }
						}
					}
				}
			}
		}
	}

	# Something happens - random events
	effect_group = {
		# Kids don't stay kids for long, we should fire these events reasonably regularly
		days = { 180 365 }

		trigger = {
			exists = var:hostage_target
		}

		first_valid = {
			triggered_effect = {
				trigger = { always = yes }
				effect = {
					save_scope_as = hostage_story
					var:hostage_target = {
						save_scope_as = hostage
					}
					story_owner = {
						trigger_event = {
							on_action = ongoing_hostage_loyalty_events
						}
					}
				}
			}
		}
	}

	#Maintenance group
	effect_group = {
		days = 1
		trigger = { 
			story_owner = { has_government = landless_adventurer_government }
		}
		triggered_effect = {
			trigger = { always = yes }
			effect = {
				#You became a landless adventurer! Ending story prematurely.
				story_owner = { save_scope_as = story_owner }
				debug_log = "Hostage Loyalty invalidated"
				debug_log_scopes = yes
				end_story = yes
			}
		}
	}
}
