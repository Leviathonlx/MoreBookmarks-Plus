### Debate Intents

debate_increase_movement_power_intent = {
	icon = intent_sow_discord
	scripted_animation = { animation = scheme }

	auto_complete = yes

	ai_will_do = {
		value = 100
		# You are more likely to push for your movement if it is not already favored.
		if = {
			limit = {
				situation:dynastic_cycle ?= {
					any_participant_group = {
						exists = var:movement_favored
						NOT = { participant_group_has_character = root }
					}
				}
			}
			add = 10
		}
		# You are already the leader of a movement
		if = {
			limit = { is_any_movement_leader = yes }
			add = 10
		}
	}
}

debate_gain_disciples_intent = {
	icon = scout_talent_intent
	scripted_animation = { animation = chancellor }

	auto_complete = yes

	is_valid = {
		custom_tooltip = {
			text = reached_disciple_limit
			num_of_relation_disciple < disciples_limit_value
		}
	}

	ai_will_do = {
		value = 100
		# How badly do we need new disciples?
		if = {
			limit = { num_of_relation_disciple = 0 }
			add = 50
		}
		if = {
			limit = { num_of_relation_disciple = 1 }
			add = 35
		}
		if = {
			limit = { num_of_relation_disciple = 2 }
			add = 25
		}
		# Do we have relevant skills to teach?
		if = {
			limit = {
				has_trait = confucian_education
				has_trait_xp = {
					trait = confucian_education
					value > 70
				}
			}
			add = 30
		}
		if = {
			limit = {
				merit_level >= 7
			}
			add = 25
		}
		if = {
			limit = {
				has_character_flag = passed_palace_exam
			}
			add = 25
		}
	}
}

debate_gain_elder_intent = {
	icon = scout_talent_intent
	scripted_animation = { animation = gongshou }

	auto_complete = yes

	ai_will_do = {
		value = 100
		# How badly do we need an Elder?
		if = {
			limit = {
				num_of_relation_elder = 0
			}
			add = 25
		}
		# Do we have something to learn?
		if = {
			limit = {
				has_trait = confucian_education
				has_trait_xp = {
					trait = confucian_education
					value < 40
				}
			}
			add = 30
		}
		if = {
			limit = {
				merit_level < 5
			}
			add = 25
		}
	}
}
