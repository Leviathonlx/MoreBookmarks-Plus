sacrifed_small_animal = {
	icon = public
	is_valid = {
		is_current_phase_active = yes
		any_attending_character = {
			location = scope:province
			NOT = { has_trait = compassionate }
		}
	}

	weight = {
		value = 1
		if = {
			limit = {
				scope:host = {
					OR = {
						has_trait = callous
						has_trait = sadistic
					}
				}
			}
			add = 0.5
		}
	}

	effect = {	
		scope:host = {
			save_scope_as = root_scope
		}
		
		random_attending_character = {
			limit = {
				location = scope:province
				NOT = { has_trait = compassionate }
			}
			save_scope_as = first
		}
		
		add_activity_log_entry = {
			key = sacrifed_small_animal
			tags = { pulse_action }
			character = scope:first
			
			scope:first = {
				add_stress = minor_stress_loss
			}
		}
	}
}

			count >= 2
		}
	}

	weight = {
		value = 1
	}

	effect = {			
		random_attending_character = {
			limit = {
				location = scope:province
			}
			save_scope_as = first
		}
		
		random_attending_character = {
			limit = {
				location = scope:province
				this != scope:first
			}
			save_scope_as = second
		}
		
		add_activity_log_entry = {
			key = praised_horned_god_together
			tags = { pulse_action }
			character = scope:first
			
			scope:first = {
				add_opinion = {
					target = scope:second
					modifier = friendliness_opinion
					opinion = 10
				}
			}
		}
	}
}

apa_imbued_artifact = {
	icon = intrigue
	is_valid = {
		is_current_phase_active = yes
		any_attending_character = {
			location = scope:province
			any_character_artifact = {
				category = inventory
				NOT = { has_variable = had_apa_imbued_artifact_var }
			}
		}
	}

	weight = {
		value = 1
	}

	effect = {			
		random_attending_character = {
			limit = {
				location = scope:province
				any_character_artifact = {
					category = inventory
					NOT = { has_variable = had_apa_imbued_artifact_var }
				}
			}
			save_scope_as = first
		}
		
		scope:first = {
			random_character_artifact = {
				limit = {
					category = inventory
					NOT = { has_variable = had_apa_imbued_artifact_var }
				}
				weight = {
					base = 1
					modifier = {
						rarity = common
						add = 2
					}
				}
				save_scope_as = artifact
				set_variable = {
					name = had_apa_imbued_artifact_var
					years = 2
				}
			}
		}
		
		add_activity_log_entry = {
			key = apa_imbued_artifact
			tags = { pulse_action }
			character = scope:first
			
			scope:artifact = {
				add_artifact_modifier = artifact_monthly_piety_gain_mult_1_modifier
			}
		}
	}
}
