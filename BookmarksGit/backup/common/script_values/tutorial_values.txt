hint_title_importance_value = {
	#direct neigbour
	if = {
		limit = {
			scope:title = {
				is_neighbor_to_realm = root
			}
		}
		add = 2
	}
	#de jure
	if = {
		limit = {
			any_held_title = {
				is_de_jure_liege_or_above_target = scope:title
			}
		}
		add = 5
	}
	if = {
		limit = {
			any_held_title = {
				target_is_de_jure_liege_or_above = scope:title
			}
		}
		add = 5
	}
	# capital
	if = {
		limit = {
			scope:title.duchy.title_capital_county = scope:title
		}
		add = 1
	}
	#claim
	if = {
		limit = {
			has_claim_on = scope:title
		}
		add = 10
	}
	#holy site
	if = {
		limit = {
			scope:title.tier = tier_county
			scope:title = {
				any_county_province = {
					barony = {
						is_holy_site_of = root.faith
					}
				}
			}
		}
		add = {
			if = {
				limit = {
					scope:title.holder.faith != root.faith
				}
				value = 10
			}
			else = {
				value = 5
			}
		}
	}
	#different faith
	if = {
		limit = {
			faith != scope:title.holder.faith
		}
		add = 5
	}
	#different culture
	if = {
		limit = {
			culture = scope:title.holder.culture
		}
		add = 5
	}
}
hint_war_importance_value = {
	if = {
		limit = {
			is_at_war = yes
		}
		add = -10
	}
	if = {
		limit = {
			current_military_strength <= scope:title.holder.top_liege.current_military_strength
		}
		add = -5
	}
	if = {
		limit = {
			max_military_strength > {
				value = scope:title.holder.top_liege.current_military_strength
				multiply = 2
			}
		}
		add = 10
	}
	if = {
		limit = {
			is_character_interaction_potentially_accepted = {
				recipient = scope:title.holder
				interaction = offer_vassalization_interaction
			}
		}
		add = -10
	}
}
