debate_outcome_value = {
	value = 0
    if = {
    	limit = {
	    	exists = var:challenged_movement_leader
	    	exists = var:debate_contender
    	}
	    if = {
	        limit = {
	           var:challenged_movement_leader.var:tgp_debate_success_score >= { value = var:debate_contender.var:tgp_debate_success_score multiply = 2 }
	        }
	        add = -3
	    }
	    else_if = {
	        limit = {
	            var:debate_contender.var:tgp_debate_success_score >= { value = var:challenged_movement_leader.var:tgp_debate_success_score multiply = 2 }
	        }
	    	add = 3
	    }
	    else_if = {
	        limit = {
	            var:challenged_movement_leader.var:tgp_debate_success_score >= { value = var:debate_contender.var:tgp_debate_success_score multiply = 1.5 }
	        }
	    	add = -2
	    }
	    else_if = {
	        limit = {
	            var:debate_contender.var:tgp_debate_success_score >= { value = var:challenged_movement_leader.var:tgp_debate_success_score multiply = 1.5 }
	        }
	    	add = 2
	    }
	    else_if = {
	        limit = {
	            var:challenged_movement_leader.var:tgp_debate_success_score >= { value = var:debate_contender.var:tgp_debate_success_score multiply = 1.2 }
	        }
	    	add = -1
	    }
	    else_if = {
	        limit = {
	            var:debate_contender.var:tgp_debate_success_score >= { value = var:challenged_movement_leader.var:tgp_debate_success_score multiply = 1.2 }
	        }
	    	add = 1
	    }
	}
}
