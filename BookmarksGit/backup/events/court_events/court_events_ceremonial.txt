namespace = court_ceremonial

scripted_effect court_ceremonial_select_bribes_effect = {
	if = { # Spend favor
		limit = {
			scope:petition_vassal = { has_hook = scope:petition_liege }
		}
		save_scope_value_as = { name = repay_favor value = yes }
	}
	else = {
		if = { # Blocs
			limit = {
				needs_more_valid_compensation_options_trigger = yes
				exists = house.house_confederation
				NOT = { scope:petition_vassal.house.house_confederation ?= house.house_confederation }
			}
			save_scope_value_as = { name = bribe_bloc value = yes }
		}
		if = { # Favor hook counter
			limit = {
				needs_more_valid_compensation_options_trigger = yes
				can_add_hook = { target = scope:petition_vassal type = favor_hook }
			}
			save_scope_value_as = { name = bribe_favor value = yes }
		}
		if = { # Gold
			limit = {
				needs_more_valid_compensation_options_trigger = yes
				scope:petition_vassal.gold >= scope:petition_vassal.medium_gold_value
			}
			save_scope_value_as = { name = bribe_gold value = yes }
		}
		if = { # Knight
			limit = {
				needs_more_valid_compensation_options_trigger = yes
				scope:petition_vassal = {
					any_close_family_member = { petition_liege_knight_trigger = yes }
				}
			}
			scope:petition_vassal = {
				random_close_family_member = {
					limit = { petition_liege_knight_trigger = yes }
					weight = {
						base = 100
						modifier = {
							add = {
								value = prowess
								multiply = 2
							}
						}
						modifier = {
							add = {
								value = age
								multiply = -1
							}
						}
						modifier = {
							factor = 2
							is_child_of = scope:petition_vassal
						}
					}
					save_scope_as = knight
				}
			}
			save_scope_value_as = { name = bribe_knight value = yes }
		}
		if = { # Vow
			limit = {
				needs_more_valid_compensation_options_trigger = yes
				scope:petition_vassal = {
					any_close_family_member = { petition_liege_vows_trigger = yes  }
				}
			}
			scope:petition_vassal = {
				random_close_family_member = {
					limit = { petition_liege_vows_trigger = yes }
					weight = {
						base = 100
						modifier = {
							add = {
								value = learning
								multiply = 2
							}
						}
						modifier = {
							add = {
								value = age
								multiply = -1
							}
						}
						modifier = {
							factor = 2
							is_child_of = scope:petition_vassal
						}
					}
					save_scope_as = monk
				}
				save_scope_value_as = { name = bribe_vows value = yes }
			}
		}
	}
}

scripted_effect court_ceremonial_petition_vassal_effect = {
	save_scope_as = petition_vassal
	set_variable = {
		name = petition_type
		value = flag:$TYPE$
	}
	add_character_flag = {
		flag = petition_ceremonial_liege_cooldown
		years = 10
	}
	court_event_character_flag_effect = yes
}

scripted_effect court_ceremonial_repay_favor_effect = {
	scope:petition_vassal = {
		remove_hook = { target = root }
	}
	petition_liege_0121_$TYPE$_effect = yes
	petition_pos_grandeur_renown_effect = yes
}

scripted_effect court_ceremonial_bloc_bribe_effect = {
	random_list = {
		10 = {
			show_chance = no
			desc = petition_liege.0101.bribe_given
			scope:petition_vassal.house = {
				tgp_join_house_bloc_effect = {
					INVITER = root.house
					OPINION = flag:yes
				}
			}
			petition_liege_0121_$TYPE$_effect = yes
			petition_pos_grandeur_renown_effect = yes
		}
		10 = {
			show_chance = no
			desc = petition_liege.0101.bribe_refused
			reverse_add_opinion = {
				modifier = disappointed_opinion
				target = scope:petition_vassal
				opinion = -25
			}
		}
	}
	stress_impact = {
		vengeful = miniscule_stress_impact_loss
		arbitrary = miniscule_stress_impact_loss
		irritable = minor_stress_impact_loss
		forgiving = medium_stress_impact_gain
		just = major_stress_impact_gain
	}
}

scripted_trigger court_ceremonial_1000_vassal_trigger = {
	has_court_event_flag = no
	is_ai = yes
	government_allows = administrative
	is_house_head = yes
	any_held_title = { is_noble_family_title = yes }
	has_succession_appointment_investors = top_liege.primary_title
	NOR = {
		house = root.house
		this = top_liege.primary_title.current_heir
		has_any_bad_relationship_with_character_trigger = { CHARACTER = root }
		has_character_flag = petition_ceremonial_liege_cooldown
		exists = house.var:petition_liege_regent_support
	}
}
