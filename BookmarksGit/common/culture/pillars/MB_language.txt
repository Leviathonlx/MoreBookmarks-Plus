#MB/CE/EPE Language File
#I generally broke languages down further than CE and attempt to make languages more consistently based on the lowest level of language families rather than the inconsistent way of vanilla and CE. For example rather than the broad West and East Iranian which covers a ton of non mutually intelligible languages I break them down further

#Generally organized by language family mostly since language 'branches' are incredibly inconsistent (ie 'hellenic' language branch being applied to arberian)

################################################################## HISTORICAL #############################################################################

language_phoenician = { #
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_phoenician
		}
	}
	parameters = {
		language_branch_semitic = yes
		language_family_afro_asiatic = yes 		
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_phoenician }
			multiply = 10
		}
		else_if = {
			limit = { has_cultural_parameter = language_branch_semitic }
			multiply = 5
		}
		else_if = {
			limit = { has_cultural_parameter = language_family_afro_asiatic }
			multiply = 2.5
		}
	}
	
	color = { 0.15 0.18 0.62 }
}
