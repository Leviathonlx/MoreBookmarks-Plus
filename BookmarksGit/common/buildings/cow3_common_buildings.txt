					}

						modifier = tunnel_03_modifier
					}

				limit = {
					has_character_modifier = tunnel_02_modifier
				}
					remove_character_modifier =	tunnel_02_modifier
					
					add_character_modifier = {
						modifier = tunnel_03_modifier
					}
			}
		}
