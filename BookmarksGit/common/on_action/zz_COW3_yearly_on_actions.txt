# Called every Jan 1st
# Root does not exist


yearly_global_pulse = {	
	on_actions = {
		zz_upgrade_cow_on_actions
	}
}

zz_upgrade_cow_on_actions = {	
	effect = {
		if = {
			limit = {
				has_game_rule = use_backgrounds
			}
			every_province = {
				limit = {
					county = {
						development_level >= 25
					}
					is_county_capital = yes
				}

				save_scope_as = new_barony

				if = {
					limit = {
						has_building = castle_01
					}
					add_building = castle_02
				}

				scope:new_barony = {
					county = {
						save_scope_value_as = {
							name = newest_development_level
							value = {
								add = development_level
							}
						}
					}
				}

				county.holder = {
					
					trigger_event = {
						id = populate_province.003
						days = 1
					}
				}
			}
		}
		if = {
			limit = {
				has_game_rule = dont_use_backgrounds
			}
			every_province = {
				limit = {
					county = {
						development_level >= 25
					}
					is_county_capital = yes
				}

				save_scope_as = new_barony

				scope:new_barony = {
					county = {
						save_scope_value_as = {
							name = newest_development_level
							value = {
								add = development_level
							}
						}
					}
				}

				county.holder = {
					
					trigger_event = {
						id = populate_province.004
						days = 1
					}
				}
			}
		}
	}
}
