			}

			}

				has_title_law = royal_elective_succession_law
				has_title_law = kurultai_elective_succession_law
			}
		}
	}
	effect = {
		every_held_title = {
			limit = { has_title_law = outremer_elective_succession_law }
			clear_title_laws = yes
		}
