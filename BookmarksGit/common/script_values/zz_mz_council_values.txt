## Eldership councilor modifiers
mz_eldership_chancellor_scale = {
	value = 0
	if = {
		limit = { # Both liege and councilor must have this doctrine parameter for this to work
			culture = { has_cultural_parameter = mz_eldership_councilor_bonus }
			liege = { 
				culture = { has_cultural_parameter = mz_eldership_councilor_bonus }
			}
			age >= 50
		}
		add = 1
		floor = yes
		min = 1
	}
}
