cow_vineyards = {
	value =  0
	every_directly_owned_province = {
		limit = {
			AND = {
				has_holding_type = castle_holding
				NOR = {
					has_building = vineyard_01
					has_building = tunnel_01
					has_building = hospital_01
					has_building = militia_01
				}
				OR = {
					AND = {
						building_slots <= 5
						has_free_building_slot = yes
					}
					AND = {
						building_slots <= 5
						has_free_building_slot = no
						#has_game_rule = yes_add_slots
					}
					AND = {
						has_free_building_slot = yes
						building_slots > 5
					}
				}
				NOT = { has_building_or_higher = vineyard_01 }
				OR = {
					terrain = drylands
					terrain = dry_hills
					terrain = hills
					terrain = farmlands
					terrain = plains
					terrain = floodplains
				}
			}
		}
		add = 1
	}
	every_directly_owned_province = {
		limit = {
			AND = {
				has_holding_type = city_holding
				NOR = {
					has_building = vineyard_01
					has_building = tunnel_01
					has_building = hospital_01
					has_building = militia_01
				}
				OR = {
					AND = {
						building_slots <= 5
						has_free_building_slot = yes
					}
					AND = {
						building_slots <= 5
						has_free_building_slot = no
						#has_game_rule = yes_add_slots
					}
					AND = {
						has_free_building_slot = yes
						building_slots > 5
					}
				}
				NOT = { has_building_or_higher = vineyard_01 }
				OR = {
					terrain = drylands
					terrain = dry_hills
					terrain = hills
					terrain = farmlands
					terrain = plains
					terrain = floodplains
				}
			}
		}
		add = 1
	}
	every_directly_owned_province = {
		limit = {
			has_holding_type = church_holding
			barony = {
				NOT = { exists = lessee }
			}
				NOR = {
					has_building = vineyard_01

					has_building = hospital_01
					has_building = militia_01
				}
			OR = {

					}

				OR = {
					terrain = drylands
					terrain = dry_hills
					terrain = hills
