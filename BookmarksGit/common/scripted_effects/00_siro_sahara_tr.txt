## siro_e_sh_remove_modifier = remove old modifier

siro_e_yearly_revaluate_sh = {
	change_global_variable = { ## keep this value untill modifier changes, if this value reachs interval, modifier will be changed.
		name = siro_g_sh_val_c_keep
		add = global_var:siro_g_sh_val_c
	}	
	if = {
		limit = {
			OR = {
				AND = {
					global_var:siro_g_sh_val = { 
						this < 20
					}
					global_var:siro_g_sh_val_c_keep = {
						OR = {
							this >= 4
							this <= -4
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						this >= 20
						this < 29
					}
					global_var:siro_g_sh_val_c_keep = {
						OR = {
							this >= 3
							this <= -3
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						this >= 29
						this < 35
					}
					global_var:siro_g_sh_val_c_keep = {
						OR = {
							this >= 2
							this <= -2
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						this >= 35
						this < 39
					}
					global_var:siro_g_sh_val_c_keep = {
						OR = {
							this >= 1
							this <= -1
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						this >= 39
						this < 69
					}
					global_var:siro_g_sh_val_c_keep = { 
						OR = {
							this >= 2
							this <= -2
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						this >= 69
						this < 111
					}
					global_var:siro_g_sh_val_c_keep = { 
						OR = {
							this >= 3
							this <= -3
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						this >= 111
						this < 171
					}
					global_var:siro_g_sh_val_c_keep = { 
						OR = {
							this >= 4
							this <= -4
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						this >= 171
						this < 246
					}
					global_var:siro_g_sh_val_c_keep = { 
						OR = {
							this >= 5
							this <= -5
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						this >= 246
						this < 336
					}
					global_var:siro_g_sh_val_c_keep = { 
						OR = {
							this >= 6
							this <= -6
						}
					}
				}
				AND = {
					global_var:siro_g_sh_val = { 
						this >= 336
						this < 413
					}
					global_var:siro_g_sh_val_c_keep = { 
						OR = {
							this >= 7
							this <= -7
						}
					}
				}
			}
		}
		change_global_variable = {
			name = siro_g_sh_val
			add = global_var:siro_g_sh_val_c
		}	
		siro_e_sh_remove_modifier = yes
		siro_e_yearly_global_sh_modifier = yes
		
		set_global_variable = {
			name = siro_g_sh_val_c_keep
			value = 0
		}
		set_global_variable = {
			name = siro_g_sh_val_c
			value = 0
		}
	}
	else = {
		change_global_variable = {
			name = siro_g_sh_val
			add = global_var:siro_g_sh_val_c
		}	
		set_global_variable = {
			name = siro_g_sh_val_c
			value = 0
		}
	}
}
