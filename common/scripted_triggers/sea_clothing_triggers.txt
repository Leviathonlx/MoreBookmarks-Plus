## Overwriting Vanilla's ##

portrait_indian_clothing_contents_trigger = {
    OR = {
        AND = {
            cfp_is_loaded = yes
            has_clothing_gfx = indian_clothing_gfx
            NOT = { has_clothing_gfx = austronesian_clothing_gfx }
        }
        AND = {
            cfp_is_loaded = no
            has_clothing_gfx = indian_clothing_gfx
        }
    }
}

## CHINA ##

portrait_chinese_clothing_trigger = {
	OR = {
		culture = {
			OR = {
				has_clothing_gfx = chinese_clothing_gfx
				has_clothing_gfx = han_clothing_gfx
			}
		}
		any_liege_or_above = {
			culture = {
				OR = {
					has_clothing_gfx = chinese_clothing_gfx
					has_clothing_gfx = han_clothing_gfx
				}
			}
		}
	}
}

portrait_chinese_wear_armor_trigger = {
	OR = {
		AND = {
			is_ruler = yes
			has_government = mercenary_government
		}
		is_commanding_army = yes
		AND = {
			is_knight = yes
			is_councillor = no
			OR = {
				is_landed = no
				liege = {
					is_at_war = yes
				}
			}
		}
		AND = {
			has_council_position = councillor_marshal
			OR = {
				is_landed = no
				primary_title.tier < tier_empire
			}
		}
	}
}

portrait_chinese_wear_helmet_trigger = {
	OR = {
		AND = {
			is_ruler = yes
			has_government = mercenary_government
		}
		AND = {
			is_ruler = yes
			has_government = holy_order_government
		}
		has_trait = order_member
		is_commanding_army = yes
		AND = {
			is_knight = yes
			is_councillor = no
			OR = {
				is_landed = no
				AND = {
					exists = primary_title
					primary_title.tier < tier_empire
					liege = {
						is_at_war = yes
					}
				}
			}
		}
		AND = {
			has_council_position = councillor_marshal
			OR = {
				is_landed = no
				primary_title.tier < tier_empire
			}
		}
	}
}
