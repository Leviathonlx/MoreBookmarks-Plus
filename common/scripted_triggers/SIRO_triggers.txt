siro_tc_has_potala = {  ## scope:county
	OR = {
		holder = {		
			title:b_potala = {
				holder = prev
				title_province = {
					OR = {
						has_building = siro_b_potala_palace_01
						has_building = siro_b_potala_palace_02
					}
				}
			}
		}
		holder.top_liege = {
			title:b_potala = {
				holder = prev
				title_province = {
					OR = {
						has_building = siro_b_potala_palace_01
						has_building = siro_b_potala_palace_02
					}
				}
			}
		}
	}
}
