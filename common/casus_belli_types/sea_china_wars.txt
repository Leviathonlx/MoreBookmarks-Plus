			}
			desc = liberty_war_victory_desc
		}
		desc = liberty_war_victory_end_desc

	}

	on_victory = {
		scope:attacker = { show_pow_release_message_effect = yes }
		scope:defender = {
			if = {
				limit = { has_realm_law = bureaucratic_authority_1 }
				add_realm_law = bureaucratic_authority_0
			}
			if = {
				limit = { has_realm_law = bureaucratic_authority_2 }
				add_realm_law = bureaucratic_authority_1
			}
			if = {
				limit = { has_realm_law = bureaucratic_authority_3 }
				add_realm_law = bureaucratic_authority_2
			}
			if = {
				limit = { has_realm_law = bureaucratic_authority_4 }
				add_realm_law = bureaucratic_authority_3
			}
			if = {
				limit = { has_realm_law = bureaucratic_authority_5 }
				add_realm_law = bureaucratic_authority_4
			}
			add_prestige = -500

			every_character_war = {
				limit = {
					is_attacker = scope:attacker
					is_defender = scope:defender
				}
				every_war_attacker = {
					if = {
						limit = {
							can_add_hook = {
								type = favor_hook
								target = scope:defender
							}
						}
						add_hook = {
							type = favor_hook
							target = scope:defender
						}
					}
				}
			}

			add_character_flag = {
				flag = recent_reduce_imperial_authority_faction_war
				years = imperial_liberty_war_victory_cooldown
			}
		}

		hidden_effect = {
			scope:attacker = {
				add_truce_both_ways = {
					character = scope:defender

					war = root.war
					result = victory
				}
				joined_faction = {
					save_scope_as = saved_faction
					add_faction_discontent = -200
					every_faction_member = {
						if = {
							limit = {
								exists = scope:attacker.joined_faction # Can get destroyed as we loop through
							}
							leave_faction_with_cooldown_effect = {
								FACTION = scope:attacker.joined_faction
								YEARS = imperial_liberty_war_victory_cooldown
							}
						}
						else = {
							add_faction_cooldown_effect = { YEARS = imperial_liberty_war_victory_cooldown }
						}
					}
				}
				# The faction should have already been destroyed due to all members leaving above, but in case it hasn't, destroy it now.
				if = {
					limit = { exists = scope:saved_faction }
					scope:saved_faction = {
						destroy_faction = yes
					}
				}
			}
		}
	}
