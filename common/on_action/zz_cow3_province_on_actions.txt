lose_health_on_actions = {
	trigger = {
		has_character_modifier = hospital_health_boost_modifier
		NOT = {
			any_directly_owned_province = {
				AND = {
					is_county_capital = yes
					OR = {
						has_holding_type = castle_holding
						AND = { 
							has_holding_type = city_holding
							barony = {
								NOT = { exists = lessee }
							}
						}	
						AND = { 
							has_holding_type = church_holding
							barony = {
								NOT = { exists = lessee }
							}
						}	
					}
					has_building_or_higher = hospital_02
				}
			}
		}
	}
	effect = {
		remove_character_modifier = hospital_health_boost_modifier		
	}
}

update_provinces_on_actions = {
	effect = {
		if = {
			limit = { 
				NOT = { 
					is_target_in_global_variable_list = {
						name = unavailable_unique_decisions
						target = flag:populated_provinces
					}
				}
			}
			every_ruler = {
				limit = {
					is_ai = no
					NOT = { has_character_flag = populated_provinces_done }
				}
				trigger_event = populate_province.001
			}
		}
	}
}
