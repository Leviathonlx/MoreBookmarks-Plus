	possible = {
		has_fp2_dlc_trigger = yes
		NOT = {
			is_target_in_global_variable_list = {
				name = unavailable_achievements
				target = flag:rd_character_blocked_iberia_or_iberia_achievement
			}
		}
	}
	happened = {
		custom_description = {
			text = fp2_iberia_or_iberia_trigger

	}
}
