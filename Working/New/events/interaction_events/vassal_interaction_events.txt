#Vassal selects response
vassal_interaction.0001 = {
	type = letter_event
	sender = scope:actor
	opening = vassal_interaction.0001.opening
	desc = vassal_interaction.0001.desc
	
	immediate = {
		scope:target = {
			primary_defender = {
				save_scope_as = defender
			}
		}
		random = { #Roll to see whether they accept at all
			chance = stop_war_attacker_likelihood_calculation
			random_list = { #Roll to see if they make demands
				5 = {
					modifier = {
						add = 10
						has_trait = craven
					}
					modifier = {
						add = 5
						has_trait = humble
					}
					modifier = {
						add = 5
						has_trait = content
					}
					modifier = {
						# Stubborn characters will always ask for something UNLESS they are intimidated.
						factor = 0
						has_trait = stubborn
						NOT = {
							has_dread_level_towards = {
								target = scope:actor
								level = 1
							}
						}
					}
					modifier = {
						# Intimidated characters more likely to accept without asking for anything.
						factor = 2
						has_dread_level_towards = {
							target = scope:actor
							level = 1
						}
					}
					save_scope_value_as = {
						name = response
						value = flag:accept
					}
				}
				5 = {
					modifier = {
						add = 10
						has_trait = ambitious
					}
					modifier = {
						# Cowed characters never ask for anything.
						factor = 0
						has_dread_level_towards = {
							target = scope:actor
							level = 2
						}
					}
					save_scope_value_as = {
						name = response
						value = flag:favor
					}
				}
				5 = {
					modifier = {
						add = 10
						has_trait = greedy
					}
					modifier = {
						# Cowed characters never ask for anything.
						factor = 0
						has_dread_level_towards = {
							target = scope:actor
							level = 2
						}
					}
					save_scope_value_as = {
						name = response
						value = flag:gold
					}
				}
			}
		}
		if = {
			limit = {
				NOT = { exists = scope:response }
			}
			save_scope_value_as = {
				name = response
				value = flag:refuse
			}
		}
	}

	option = { #Accept
		name = vassal_interaction.0001.a
		show_as_tooltip = {
			scope:target = {
				end_war = white_peace
			}
			scope:actor = {
				add_opinion = {
					target = root
					modifier = pleased_opinion
					opinion = low_positive_opinion
				}
			}
		}
		scope:actor = {
			trigger_event = {
				id = vassal_interaction.0002
				days = { 3 6 }
			}
		}
		ai_chance = {
			base = 0
			modifier = {
				add = 100
				scope:response = flag:accept
			}
		}
	}

	option = { #Request a favor
		name = vassal_interaction.0001.b
		trigger = {
			can_add_hook = {
				target = scope:actor
				type = favor_hook
			}
		}
		custom_tooltip = vassal_interaction.0001.request_favor_tt
		show_as_tooltip = {
			random_list = {
				1 = {
					desc = vassal_interaction.0001.accept
					show_chance = no
					scope:target = {
						end_war = white_peace
					}
					add_hook = {
						target = scope:actor
						type = favor_hook
					}
				}
				1 = {
					desc = vassal_interaction.0001.decline
					show_chance = no
					add_prestige = medium_prestige_loss
				}
			}
		}
		scope:actor = {
			trigger_event = {
				id = vassal_interaction.0003
				days = { 3 6 }
			}
		}
		ai_chance = {
			base = 0
			modifier = {
				add = 100
				scope:response = flag:favor
			}
		}
	}
	
	option = { #Request gold
		name = vassal_interaction.0001.c
		custom_tooltip = vassal_interaction.0001.request_gold_tt
		show_as_tooltip = {
			random_list = {
				1 = {
					desc = vassal_interaction.0001.accept
					show_chance = no
					scope:target = {
						end_war = white_peace
					}
					scope:actor = {
						pay_short_term_gold = {
							target = root
							gold = scope:recipient.stop_war_attacker_bribe_size
						}
					}
				}
				1 = {
					desc = vassal_interaction.0001.decline
					show_chance = no
					add_prestige = medium_prestige_loss
				}
			}
		}
		scope:actor = {
			trigger_event = {
				id = vassal_interaction.0004
				days = { 3 6 }
			}
		}
		ai_chance = {
			base = 0
			modifier = {
				add = 100
				scope:response = flag:gold
			}
		}
	}
	
	option = { #Refuse
		name = vassal_interaction.0001.d
		trigger = {
			NOT = { has_usable_hook = scope:actor }
		}
		show_as_tooltip = {
			scope:actor = {
				add_opinion = {
					target = root
					modifier = refusal_opinion
					opinion = low_negative_opinion
				}
			}
		}
		scope:actor = {
			trigger_event = {
				id = vassal_interaction.0005
				days = { 3 6 }
			}
		}
		ai_chance = {
			base = 0
			modifier = {
				add = 100
				scope:response = flag:refuse
			}
		}
	}

	option = { #Refuse - you have a hook!
		name = vassal_interaction.0001.e
		trigger = {
			has_usable_hook = scope:actor
		}
		show_as_tooltip = {
			use_hook = scope:actor
		}
		save_scope_value_as = {
			name = vassal_used_hook
			value = yes
		}
		scope:actor = {
			trigger_event = {
				id = vassal_interaction.0005
				days = { 3 6 }
			}
		}
		ai_chance = {
			base = 0
			modifier = {
				add = 100
				scope:response = flag:refuse
			}
		}
	}
}


#Liege: Vassal accepts
vassal_interaction.0002 = {
	type = letter_event
	sender = scope:recipient
	opening = vassal_interaction.0002.opening
	desc = vassal_interaction.0002.desc
	
	immediate = {
		show_as_tooltip = {
			scope:target = {
				end_war = white_peace
			}
		}
		hidden_effect = {
			save_scope_as = war_ender
			send_interface_toast = {
				title = vassal_interaction.war_ender_gives_up
				scope:target = {
					end_war = white_peace
				}
			}
			add_opinion = {
				target = scope:recipient
				modifier = pleased_opinion
				opinion = low_positive_opinion
			}
		}
	}

	option = {
		name = vassal_interaction.0002.a
		scope:defender = {
			add_opinion = {
				target = root
				modifier = grateful_opinion
				opinion = 10
			}
		}
		
		# If we're a clan this event affects unity
		stop_attacker_vassal_war_add_clan_unity_effect = yes
	}
}


#Liege: Vassal asks for a favor
vassal_interaction.0003 = {
	type = letter_event
	opening = vassal_interaction.0002.opening
	desc = vassal_interaction.0003.desc
	sender = {
		character = scope:recipient
		animation = personality_greedy
	}

	option = { #Very well
		name = vassal_interaction.0003.a
		show_as_tooltip = {
			scope:target = {
				end_war = white_peace
			}
			scope:recipient = {
				add_hook = {
					target = root
					type = favor_hook
				}
			}
		}
		save_scope_value_as = {
			name = reward_type
			value = flag:favor
		}
		scope:recipient = {
			trigger_event = {
				id = vassal_interaction.0006
				days = 1
			}
		}
		scope:defender = {
			add_opinion = {
				target = root
				modifier = grateful_opinion
				opinion = 10
			}
		}
		
		# If we're a clan this event affects unity
		stop_attacker_vassal_war_add_clan_unity_effect = yes
		
		ai_chance = {
			base = 100
			modifier = {
				add = 100
				has_trait = generous
			}
			ai_value_modifier = {
				ai_rationality = 0.5
				ai_honor = 0.5
			}
		}
	}

	option = { #Never!
		name = vassal_interaction.0003.b
		custom_tooltip = vassal_interaction.0003.b.tt
		scope:recipient = {
			trigger_event = {
				id = vassal_interaction.0007
				days = 1
			}
		}
		ai_chance = {
			base = 100
			modifier = {
				add = 100
				has_trait = stubborn
			}
			ai_value_modifier = {
				ai_vengefulness = 0.5
				ai_greed = 1
			}
		}
	}

	option = {
		name = vassal_interaction.you_owe_me
		trigger = {
			scope:recipient = { is_ai = yes }
			has_usable_hook = scope:recipient
		}
		use_hook = scope:recipient
		scope:target = {
			end_war = white_peace
		}
		scope:defender = {
			add_opinion = {
				target = root
				modifier = grateful_opinion
				opinion = 10
			}
		}
		
		# If we're a clan this event affects unity
		stop_attacker_vassal_war_add_clan_unity_effect = yes
	}
}

vassal_interaction.0005 = {
	type = letter_event
	sender = scope:recipient
	opening = vassal_interaction.0002.opening
	desc = vassal_interaction.0005.desc

	option = {
		name = vassal_interaction.0005.a
		trigger = {
			NOT = { exists = scope:vassal_used_hook }
		}
		add_prestige = minor_prestige_loss
		hidden_effect = {
			add_opinion = {
				target = scope:recipient
				modifier = refusal_opinion
				opinion = low_negative_opinion
			}
		}
	}

	option = {
		name = vassal_interaction.you_owe_me
		trigger = {
			scope:recipient = { is_ai = yes }
			has_usable_hook = scope:recipient
			NOT = { exists = scope:vassal_used_hook }
		}
		use_hook = scope:recipient
		scope:target = {
			end_war = white_peace
		}
	}

	option = {
		name = vassal_interaction.0005.b
		trigger = {
			exists = scope:vassal_used_hook
		}
		scope:recipient = {
			use_hook = scope:actor
		}
	}
}
