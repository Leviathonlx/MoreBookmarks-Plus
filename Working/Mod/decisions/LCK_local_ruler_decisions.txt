found_local_kingdom_decision = {
	picture = "gfx/interface/illustrations/decisions/decision_view_cow_building.dds"
	major = yes
	ai_check_interval = 600
	desc = found_local_kingdom_decision_desc

	is_shown = {
		is_ai = no
		highest_held_title_tier <= tier_duchy
		is_independent_ruler = yes
	}

	is_valid = {		
		is_independent_ruler = yes
		highest_held_title_tier = tier_duchy
		
		root.capital_province = {
			AND = {
				has_holding_type = castle_holding
				has_building_or_higher = castle_05
				
			}
		}
	}
	
	is_valid_showing_failures_only = {
		is_available_adult = yes
		is_at_war = no
	}

	effect = {
		create_custom_kingdom_effect = yes
		if = {
			limit = {
				is_ai = no
				NOT = { exists = global_var:found_kingdom_decision  }
			}
			set_global_variable = {
				name = found_kingdom_decision
				value = root
			}
			set_global_variable = {
				name = found_kingdom_decision_kingdom
				value = root.primary_title
			}
		}
	}

	cost = {
		gold = 300
		prestige = 500
		piety = 200
	}

	ai_potential = {
		always = no
	}

	ai_will_do = {
		base = -100
	}
}
