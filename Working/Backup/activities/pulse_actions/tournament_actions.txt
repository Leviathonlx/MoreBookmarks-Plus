		random_attending_character = {
			limit = { 
				has_trait = tourney_participant

			}
			save_scope_as = first
		}

		add_activity_log_entry = {
			key = tournament_studied_techniques
			tags = { pulse_action }
			character = scope:first

			scope:first = {
				random_list = {
					1 = {
						add_trait_xp = {
							trait = tourney_participant
							track = foot
							value = {
								integer_range = {
									min = small_lifestyle_random_xp_low
									max = small_lifestyle_random_xp_high
								}
							}
						}
					}
					1 = {
						add_trait_xp = {
							trait = tourney_participant
							track = bow
							value = {
								integer_range = {
									min = small_lifestyle_random_xp_low
									max = small_lifestyle_random_xp_high
								}
							}
						}
					}
					1 = {
						add_trait_xp = {
							trait = tourney_participant
							track = horse
							value = {
								integer_range = {
									min = small_lifestyle_random_xp_low
									max = small_lifestyle_random_xp_high
								}
							}
						}
					}
					1 = {
						add_trait_xp = {
							trait = tourney_participant
							track = wit
							value = {
								integer_range = {
									min = small_lifestyle_random_xp_low
									max = small_lifestyle_random_xp_high
								}
							}

					}

						# Add Blademaster Trait XP - only if you are a blademaster

		scope:host = {
			save_scope_as = first
		}
		random_guest_subset = {
			name = contestant

					target = scope:host

				}

		}

				}
			}
		}
	}
