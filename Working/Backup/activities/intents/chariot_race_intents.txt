# Default intent for all guests
increase_influence_intent = {
	icon = intent_influence

	auto_complete = yes

	on_invalidated = { trigger_event = activity_system.0090 }

	ai_will_do = {
		value = 100
		if = {
			limit = { is_landless_adventurer = yes }
			add = -500
		}
	}

	scripted_animation = {
		animation = personality_honorable
	}
}

# Available to guests only. Aimed at lowering popular opinion of the host in the capital
sow_discord_intent = {
	icon = intent_sow_discord

	is_shown = {
		is_alive = yes
	}

	ai_will_do = {
		value = 0
		# TRAITS
		if = {
			limit = { has_trait = ambitious }
			add = 50
		}
		if = {
			limit = { has_trait = disloyal }
			add = 50
		}
		if = {
			limit = { has_trait = sadistic }
			add = 50
		}
		if = {
			limit = { has_trait = deceitful }
			add = 25
		}
		if = {
			limit = { has_trait = arbitrary }
			add = 50
		}
		if = {
			limit = { has_trait = calm }
			subtract = 50
		}
		if = {
			limit = { has_trait = just }
			subtract = 50
		}
		if = {
			limit = { has_trait = trusting }
			subtract = 50
		}
		if = {
			limit = { has_trait = humble }
			subtract = 30
		}
		if = {
			limit = { has_trait = craven }
			subtract = 25
		}
		if = {
			limit = {
				exists = top_liege
				top_liege.capital_county = {
					county_opinion >= 10
				}
			}
			add = -25
		}
		if = {
			limit = {
				exists = top_liege
				top_liege.capital_county = {
					county_opinion < 0
				}
			}
			add = 25
		}
		if = {
			limit = {
				exists = top_liege
				top_liege.capital_county = {
					county_opinion <= -20
				}
			}
			add = 50
		}
		if = {
			limit = {
				exists = top_liege
				top_liege.capital_county = {
					county_opinion <= -40
				}
			}
			add = 50
		}
		if = {
			limit = {
				exists = top_liege
				top_liege = {
					government_allows = administrative
					culture = { has_cultural_parameter = hosts_chariot_races }
				}
				has_relation_rival = top_liege
			}
			add = 100
		}
		if = {
			limit = {
				exists = top_liege
				top_liege = {
					government_allows = administrative
					culture = { has_cultural_parameter = hosts_chariot_races }
				}
				has_relation_nemesis = top_liege
			}
			add = 200
		}
		if = {
			limit = {
				exists = top_liege
				top_liege = {
					government_allows = administrative
					culture = { has_cultural_parameter = hosts_chariot_races }
				}
				opinion = {
					target = top_liege
					value < 0
				}
			}
			add = 50
		}
		if = {
			limit = {
				exists = top_liege
				top_liege = {
					government_allows = administrative
					culture = { has_cultural_parameter = hosts_chariot_races }
				}
				opinion = {
					target = top_liege
					value <= -50
				}
			}
			add = 100
		}
		if = {
			limit = {
				exists = top_liege
				top_liege = {
					government_allows = administrative
					culture = { has_cultural_parameter = hosts_chariot_races }
				}
				opinion = {
					target = top_liege
					value <= -90
				}
			}
			add = 50
		}
		if = {
			limit = {
				exists = top_liege
				top_liege = {
					government_allows = administrative
					culture = { has_cultural_parameter = hosts_chariot_races }
				}
				opinion = {
					target = top_liege
					value >= 25
				}
			}
			add = -50
		}
		if = {
			limit = {
				exists = top_liege
				top_liege = {
					government_allows = administrative
					culture = { has_cultural_parameter = hosts_chariot_races }
				}
				opinion = {
					target = top_liege
					value >= 75
				}
			}
			add = -100
		}
		# CLAIMS
		if = {
			limit = {
				exists = top_liege
				top_liege = {
					government_allows = administrative
					culture = { has_cultural_parameter = hosts_chariot_races }
				}
				has_strong_claim_on = top_liege.primary_title
			}
			add = 100
		}
		else_if = {
			limit = {
				exists = top_liege
				top_liege = {
					government_allows = administrative
					culture = { has_cultural_parameter = hosts_chariot_races }
				}
				has_weak_claim_on = top_liege.primary_title
			}
			add = 50
		}
		if = {
			limit = {
				exists = top_liege
				top_liege = {
					government_allows = administrative
					culture = { has_cultural_parameter = hosts_chariot_races }
				}
				top_liege.primary_title = {
					any_title_heir = {
						this = root
					}
					place_in_line_of_succession = {
						target = root
						value >= 5
					}
				}
				NOT = {
					exists = top_liege.house
					house ?= top_liege.house
				}
			}
			add = 50
		}
		if = {
			limit = {
				highest_held_title_tier <= tier_county
				exists = top_liege
				top_liege = {
					government_allows = administrative
					culture = { has_cultural_parameter = hosts_chariot_races }
				}
				NOT = {
					top_liege.primary_title = {
						any_title_heir = {
							this = root
						}
						place_in_line_of_succession = {
							target = root
							value >= 20
						}
					}
				}
			}
			add = -200
		}
		if = {
			limit = { is_landless_adventurer = yes }
			add = -500
		}
	}

	scripted_animation = {
		animation = scheme
	}
}

# Available to the host (emperor) only. Aimed at increasing popular opinion in capital
appease_populace_intent = {
	icon = intent_appease_populace

	is_shown = {
		is_alive = yes
		# Only shown when you can actually improve your popular opinion in the capital
		capital_county.county_opinion <= 90
	}

	on_invalidated = { trigger_event = activity_system.0090 }

	ai_will_do = {
		value = 50
		if = {
			limit = { capital_county.county_opinion < 50 }
			add = 25
		}
		if = {
			limit = { capital_county.county_opinion < 10 }
			add = 25
		}
		if = {
			limit = { capital_county.county_opinion < 0 }
			add = 50
		}
		if = {
			limit = { capital_county.county_opinion < -10 }
			add = 100
		}
		if = {
			limit = { capital_county.county_opinion < -20 }
			add = 100
		}
		if = {
			limit = { capital_county.county_opinion < -30 }
			add = 100
		}
		if = {
			limit = { legitimacy_level >= 4 }
			add = -50
		}
		if = {
			limit = { legitimacy_level <= 2 }
			add = 50
		}
		if = {
			limit = { legitimacy_level <= 1 }
			add = 50
		}
		if = {
			limit = { legitimacy_level < 1 }
			add = 200
		}
	}

	scripted_animation = {
		animation = personality_compassionate
	}
}
